Pre-process the images for SIFT¶

The Scale Invariant Feature Transform (SIFT) algorithm requires grayscale images. In the paper, the authors also resize the images to 50 x 50.

Extract SIFT descriptors¶

In [10]:
img_descs = []

for image in resized_images:
    # SIFT complains if image is not uint8
    formatted_image = (image*255).astype('uint8')
    
    # define SIFT feature extractor
    sift = cv2.xfeatures2d.SIFT_create(nOctaveLayers=60, 
                                       contrastThreshold=0.01)
    
    # extract keypoints and descriptors
    kp, desc = sift.detectAndCompute(formatted_image, None)
    
    # if no keypoints are detected, make zeros
    if len(kp) < 1:
        desc = np.zeros((1, sift.descriptorSize()), np.float32)
        
    # the descriptors should be 128 in dimensionality
    assert desc.shape[1] == 128
    
    # add descriptor to the list
    img_descs.append(desc)
    
# concatenate all of the descriptors in the dataset
image_descriptors = np.concatenate(img_descs)

clustered_descriptors = [k_means.predict(descriptor) for descriptor in img_descs]

def cross_validation_experiment(gp):
    np.random.seed(12)

    # define error metrics
    training_times = []
    prediction_times = []
    accuracy_scores = []
    precision_scores = []
    recall_scores = []
    auc_scores = []
    f1_scores = []

    # define splits for 5-fold CV
    skf = StratifiedKFold(n_splits=5)
    skf.get_n_splits(cluster_histograms)



    # 5-fold CV loop
    for train_index, test_index in skf.split(cluster_histograms, labels):

        # split the data for training and evaluation
        X_train, X_test = cluster_histograms[train_index], cluster_histograms[test_index]

        y_train, y_test = labels[train_index], labels[test_index]
        
        # measure training time
        start_train_time = time.time()
        model_fit = gp.fit(X_train, y_train)
        training_time = time.time() - start_train_time

        # measure prediction time
        start_predict_time = time.time()
        yhat = gp.predict(X_test)
        prediction_time = time.time() - start_predict_time

        # calculate evaluation metrics
        accuracy = accuracy_score(y_test, yhat)
        precision = precision_score(y_test, yhat)
        recall = recall_score(y_test, yhat)
        auc = roc_auc_score(y_test, yhat)
        f1 = f1_score(y_test, yhat)

        ## log results from the fold
        training_times.append(training_time)
        prediction_times.append(prediction_time)
        accuracy_scores.append(accuracy)
        precision_scores.append(precision)
        recall_scores.append(recall)
        auc_scores.append(auc)
        f1_scores.append(f1)
        
    # store the results as a dataframe
    results = DataFrame({
              'accuracy':accuracy_scores,
               'precision':precision_scores,
               'recall':recall_scores,
               'roc auc score':auc_scores,
               'f1 score':f1_scores,
              'training time':training_times,
              'prediction time':prediction_times
                           })
    return results

Results¶

  • The GP with RBF kernel yielded similar results to the official implementation (slightly lower precision and recall; it's not in the 90's)

  • The GP with quadratic kernel outperformed the GP with linear kernel and the GP with RBF kernel.

  • This is most likely due to the linear decision boundary, as seen in the Isomap projections of SIFT features.
